Revealing Module Pattern

  • Creación de módulos emplando una interfaz de objeto autocontenida
  • Los métodos y variables se mantienen privados a no ser que sean explicitamente espuestos
  • Solo se declara una única variable global
  

var myModule = (function () {
// body of our module
var users = ['mohamed', 'karim', 'amine'];

return {
showUsers: function(){
users.map(function(user){
  console.log(user);
});
},
addUser: function(name){
users.push(name);
}
}
})();

myModule.showUsers();
