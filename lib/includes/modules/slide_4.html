Global imports

  • Empleado por librerías como jQuery
  • Similar al caso de anonymous closure
  • Solo se declara una única variable global
  

// global function
function showUsers(users){
users.map(function(user){
console.log(user);
});
}

(function (x) {
// body of our module
var users = ['mohamed', 'karim', 'amine'];
x(users);
})(showUsers);
