CommonJS Module Pattern

  

var users = ['mohamed', 'karim', 'amine'];

module.exports = {
showUsers: function(){
users.map(function(user){
console.log(user);
});
},
addUser: function(name){
users.push(name);
}
}

var list = require('./commonjs-module-pattern');

list.showUsers();
list.addUser('algiers tech meetup');
console.log(Array(40).join('-'));
list.showUsers();

// we can't access users.
console.log('users are : ',list.users);
